001package com.github.games647.scoreboardstats.config;
002
003import com.google.common.base.Charsets;
004import com.google.common.base.Strings;
005import com.google.common.io.Files;
006
007import java.io.File;
008import java.io.IOException;
009import java.io.InputStream;
010import java.io.InputStreamReader;
011import java.lang.reflect.Field;
012import java.lang.reflect.Modifier;
013import java.util.List;
014import java.util.logging.Level;
015
016import org.bukkit.ChatColor;
017import org.bukkit.configuration.InvalidConfigurationException;
018import org.bukkit.configuration.file.FileConfiguration;
019import org.bukkit.configuration.file.YamlConfiguration;
020import org.bukkit.plugin.Plugin;
021
022/**
023 * Represents an easy way to load and save to yaml configs. Furthermore support
024 * this class UTF-8 even before Bukkit introduced it and will also support comments
025 * soon.
026 *
027 * @param <T> the plugin type
028 */
029public class CommentedYaml<T extends Plugin> {
030
031    private static final String COMMENT_PREFIX = "# ";
032    private static final String FILE_NAME = "config.yml";
033
034    protected transient final T plugin;
035    protected transient FileConfiguration config;
036
037    public CommentedYaml(T plugin) {
038        this.plugin = plugin;
039    }
040
041    /**
042     * Gets the YAML file configuration from the disk while loading it
043     * explicit with UTF-8. This allows umlauts and other UTF-8 characters
044     * for all Bukkit versions.
045     *
046     * Bukkit add also this feature since
047     * https://github.com/Bukkit/Bukkit/commit/24883a61704f78a952e948c429f63c4a2ab39912
048     * To be allow the same feature for all Bukkit version, this method was
049     * created.
050     *
051     * @return the loaded file configuration
052     */
053    public FileConfiguration getConfigFromDisk() {
054        final File file = new File(plugin.getDataFolder(), FILE_NAME);
055
056        final YamlConfiguration newConf = new YamlConfiguration();
057        newConf.setDefaults(getDefaults());
058
059        try {
060            //UTF-8 should be available on all java running systems
061            final List<String> lines = Files.readLines(file, Charsets.UTF_8);
062
063            final StringBuilder builder = new StringBuilder();
064            for (String line : lines) {
065                //remove the silly tab error from yaml
066                builder.append(line.replace("\t", "    "));
067                //indicates a new line
068                builder.append('\n');
069            }
070
071            newConf.loadFromString(builder.toString());
072            return newConf;
073        } catch (InvalidConfigurationException ex) {
074            plugin.getLogger().log(Level.SEVERE, "Invalid Configuration", ex);
075        } catch (IOException ex) {
076            plugin.getLogger().log(Level.SEVERE, "Couldn't load the configuration", ex);
077        }
078
079        return newConf;
080    }
081
082    protected void loadConfig() {
083        if (config == null) {
084            config = getConfigFromDisk();
085        }
086
087        for (Field field : getClass().getDeclaredFields()) {
088            if (Modifier.isTransient(field.getModifiers())
089                    || (!field.getType().isPrimitive() && field.getType() != String.class)) {
090                continue;
091            }
092
093            final ConfigNode node = field.getAnnotation(ConfigNode.class);
094            String path = field.getName();
095            if (node != null && !Strings.isNullOrEmpty(path)) {
096                path = node.path();
097            }
098
099            field.setAccessible(true);
100            if (config.isSet(path)) {
101                try {
102                    if (config.isString(path)) {
103                        field.set(this, ChatColor.translateAlternateColorCodes('&', config.getString(path)));
104                    } else {
105                        field.set(this, config.get(path));
106                    }
107                } catch (IllegalAccessException ex) {
108                    plugin.getLogger().log(Level.SEVERE, null, ex);
109                }
110            } else {
111                plugin.getLogger().log(Level.INFO, "Path not fond {0}", path);
112            }
113        }
114    }
115
116    protected void saveConfig() {
117        if (config == null) {
118            return;
119        }
120
121        //todo add comment support
122        try {
123            Files.write(config.saveToString(), new File(plugin.getDataFolder(), FILE_NAME), Charsets.UTF_8);
124        } catch (IOException ex) {
125            plugin.getLogger().log(Level.SEVERE, null, ex);
126        }
127    }
128
129    /**
130     * Get the default configuration located in the plugin jar
131     *
132     * @return the default configuration
133     */
134    private FileConfiguration getDefaults() {
135        final InputStream defConfigStream = plugin.getResource(FILE_NAME);
136        if (defConfigStream != null) {
137            //stream will be closed in this method
138            return YamlConfiguration.loadConfiguration(new InputStreamReader(defConfigStream, Charsets.UTF_8));
139        }
140
141        return new YamlConfiguration();
142    }
143}
