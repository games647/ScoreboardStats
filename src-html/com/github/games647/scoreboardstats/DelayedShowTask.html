001package com.github.games647.scoreboardstats;
002
003import org.bukkit.entity.Player;
004
005/**
006 * Scheduled appear task
007 */
008public class DelayedShowTask implements Runnable {
009
010    private final SbManager scoreboardManager;
011
012    private final Player player;
013    private final boolean action;
014
015    /**
016     * Creates a new scheduled appear of the normal scoreboard or the temp scoreboard
017     *
018     * @param player the specific player
019     * @param action if the temp scoreboard be displayed
020     * @param scoreboardManager the associated sbManager
021     */
022    public DelayedShowTask(Player player, boolean action, SbManager scoreboardManager) {
023        this.scoreboardManager = scoreboardManager;
024        this.player = player;
025        this.action = action;
026    }
027
028    @Override
029    public void run() {
030        if (action) {
031            scoreboardManager.createTopListScoreboard(player);
032        } else {
033            scoreboardManager.createScoreboard(player);
034        }
035    }
036}
