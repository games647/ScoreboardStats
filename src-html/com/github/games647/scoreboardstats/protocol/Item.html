001package com.github.games647.scoreboardstats.protocol;
002
003import com.google.common.base.Objects;
004import com.google.common.base.Preconditions;
005import com.google.common.primitives.Ints;
006
007import org.apache.commons.lang.builder.ToStringBuilder;
008import org.bukkit.entity.Player;
009
010/**
011 * Represents a scoreboard item with ProtocolLib.
012 *
013 * @see Objective
014 */
015public class Item implements Comparable<Item> {
016
017    private final Objective parent;
018
019    private final String scoreName;
020    private int score;
021
022    Item(Objective parent, String scoreName, int score) {
023        this(parent, scoreName, score, true);
024    }
025
026    Item(Objective parent, String scoreName, int score, boolean send) {
027        this.parent = parent;
028
029        this.scoreName = scoreName;
030        this.score = score;
031
032        if (send) {
033            update();
034        }
035    }
036
037    /**
038     * Check if this item is shown to the player.
039     *
040     * @return whether this item is shown to the player
041     */
042    public boolean isShown() {
043        return exists() && parent.isShown();
044    }
045
046    /**
047     * Checks if this item exists client-side.
048     *
049     * @return whether this item exists
050     */
051    public boolean exists() {
052        return parent.items.containsValue(this) && parent.exists();
053    }
054
055    /**
056     * Get the unique item name.
057     *
058     * @return the unique item name
059     */
060    public String getScoreName() {
061        return scoreName;
062    }
063
064    /**
065     * Get the current score.
066     *
067     * @return the current score
068     */
069    public int getScore() {
070        return score;
071    }
072
073    /**
074     * Set the score value.
075     *
076     * @param score the new value
077     * @throws IllegalStateException if the item was removed
078     * @throws IllegalStateException if the objective was removed
079     */
080    public void setScore(int score) throws IllegalStateException {
081        Preconditions.checkState(exists(), "the parent objective or this item isn't active");
082
083        if (this.score != score) {
084            this.score = score;
085            update();
086        }
087    }
088
089    /**
090     * Unregister this item.
091     */
092    public void unregister() {
093        if (exists()) {
094            getScoreboard().resetScore(scoreName);
095            PacketFactory.sendPacket(this, State.REMOVE);
096        }
097    }
098
099    /**
100     * Get the associated parent.
101     *
102     * @return the objective that hold this item
103     */
104    public Objective getParent() {
105        return this.parent;
106    }
107
108    /**
109     * Gets the owner of this item and objective.
110     *
111     * @return the tracking player
112     */
113    public Player getOwner() {
114        return parent.getOwner();
115    }
116
117    /**
118     * Get the scoreboard instance.
119     *
120     * @return the scoreboard
121     */
122    public PlayerScoreboard getScoreboard() {
123        return parent.getScoreboard();
124    }
125
126    @Override
127    public int compareTo(Item other) {
128        //Reverse order - first the highest element like the scoreboard ingame
129        return Ints.compare(other.score, score);
130    }
131
132    @Override
133    public int hashCode() {
134        return Objects.hashCode(scoreName, score);
135    }
136
137    @Override
138    public boolean equals(Object obj) {
139        //ignores also null
140        if (obj instanceof Item) {
141            final Item other = (Item) obj;
142            return Objects.equal(scoreName, other.scoreName)
143                    && score == other.score;
144        } else {
145            return false;
146        }
147    }
148
149    @Override
150    public String toString() {
151        return ToStringBuilder.reflectionToString(this);
152    }
153
154    /**
155     * Sends a new packet
156     *
157     * @see PacketFactory
158     */
159    private void update() {
160        PacketFactory.sendPacket(this, State.CREATE);
161    }
162}
