001package com.github.games647.scoreboardstats.protocol;
002
003import org.bukkit.scoreboard.DisplaySlot;
004
005/**
006 * Represent the three different sides a scoreboard objective can have
007 *
008 * Protocol specifications can be found here http://wiki.vg/Protocol
009 */
010public class SlotTransformer {
011
012    /**
013     * Get the enum from his id
014     *
015     * @param slotId the id
016     * @return the representing enum or null if the id isn't valid
017     */
018    public static DisplaySlot fromId(int slotId) {
019        switch (slotId) {
020            case 0:
021                return DisplaySlot.PLAYER_LIST;
022            case 1:
023                return DisplaySlot.SIDEBAR;
024            case 2:
025                return DisplaySlot.BELOW_NAME;
026            default:
027                return null;
028        }
029    }
030
031    private SlotTransformer() {
032        //singleton
033    }
034}
