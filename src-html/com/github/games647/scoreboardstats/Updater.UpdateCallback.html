030public class Updater {
031
032    /* Constants */
033
034    // Remote file's title
035    private static final String TITLE_VALUE = "name";
036    // Remote file's download link
037    private static final String LINK_VALUE = "downloadUrl";
038    // Path to GET
039    private static final String QUERY = "https://api.curseforge.com/servermods/files?projectIds=";
040    // Used for locating version numbers in file names
041    private static final String DELIMETER = "^v|[\\s_-]v";
042    // If the version number contains one of these, don't update.
043    private static final String[] NO_UPDATE_TAG = {"-DEV", "-PRE", "-SNAPSHOT"};
044
045    /* User-provided variables */
046
047    // plugin running Updater
048    private final Plugin plugin;
049    // Type of update check to run
050    private final boolean shouldDownload;
051    // The plugin file (jar)
052    private final File file;
053    // The folder that downloads will be placed in
054    private final File updateFolder;
055    // The provided callback (if any)
056    private final UpdateCallback callback;
057    // Project's Curse ID
058    private int id = -1;
059
060    /* Collected from Curse API */
061
062    private String versionName;
063    private String versionLink;
064
065    /* Update process variables */
066
067    // Connection to RSS
068    private URL url;
069    // Updater thread
070    private Thread thread;
071    // Used for determining the outcome of the update process
072    private Updater.UpdateResult result = Updater.UpdateResult.SUCCESS;
073
074    /**
075     * Gives the developer the result of the update process. Can be obtained by called {@link #getResult()}
076     */
077    public enum UpdateResult {
078        /**
079         * The updater found an update, and has readied it to be loaded the next time the server restarts/reloads.
080         */
081        SUCCESS,
082        /**
083         * The updater did not find an update, and nothing was downloaded.
084         */
085        NO_UPDATE,
086        /**
087         * The updater found an update, but was unable to download it.
088         */
089        FAIL_DOWNLOAD,
090        /**
091         * For some reason, the updater was unable to contact dev.bukkit.org to download the file.
092         */
093        FAIL_DBO,
094        /**
095         * When running the version check, the file on DBO did not contain a recognizable version.
096         */
097        FAIL_NOVERSION,
098        /**
099         * The id provided by the plugin running the updater was invalid and doesn't exist on DBO.
100         */
101        FAIL_BADID,
102        /**
103         * The updater found an update, but because of the shouldDownload being set to false, it wasn't downloaded.
104         */
105        UPDATE_AVAILABLE
106    }
107
108    /**
109     * Initialize the updater.
110     *
111     * @param plugin    The plugin that is checking for an update.
112     * @param id        The dev.bukkit.org id of the project.
113     * @param file      The file that the plugin is running from, get this from getFile() from within your main class.
114     * @param download  Specify the type of update this will be.
115     */
116    public Updater(Plugin plugin, int id, File file, boolean download) {
117        this(plugin, id, file, download, null);
118    }
119
120    /**
121     * Initialize the updater with the provided callback.
122     *
123     * @param plugin    The plugin that is checking for an update.
124     * @param id        The dev.bukkit.org id of the project.
125     * @param file      The file that the plugin is running from, get this from getFile() from within your main class.
126     * @param download  Specify the type of update this will be.
127     * @param callback  The callback instance to notify when the Updater has finished
128     */
129    public Updater(Plugin plugin, int id, File file, boolean download, UpdateCallback callback) {
130        this.plugin = plugin;
131        this.shouldDownload = download;
132        this.file = file;
133        this.id = id;
134        this.updateFolder = this.plugin.getServer().getUpdateFolderFile();
135        this.callback = callback;
136
137        try {
138            this.url = new URL(Updater.QUERY + this.id);
139        } catch (MalformedURLException malformedURLEx) {
140            //This can only happend if we modified the url. Just an int cannot make it malformed
141            this.plugin.getLogger().log(Level.SEVERE, "Invalid URL", malformedURLEx);
142            this.result = UpdateResult.FAIL_BADID;
143            return;
144        }
145
146        this.thread = new Thread(new Runnable() {
147
148            @Override
149            public void run() {
150                runUpdater();
151            }
152            //Recognize our updater
153        }, plugin.getName() + "-Updater");
154
155        this.thread.start();
156    }
157
158    /**
159     * Get the result of the update process.
160     *
161     * @return result of the update process.
162     * @see UpdateResult
163     */
164    public Updater.UpdateResult getResult() {
165        this.waitForThread();
166        return this.result;
167    }
168
169    /**
170     * Get the latest version's name (such as "Project v1.0").
171     *
172     * @return latest version's name.
173     */
174    public String getLatestName() {
175        this.waitForThread();
176        return this.versionName;
177    }
178
179    /**
180     * Get the latest version's direct file link.
181     *
182     * @return latest version's file link.
183     */
184    public String getLatestFileLink() {
185        this.waitForThread();
186        return this.versionLink;
187    }

189    /**
190     * As the result of Updater output depends on the thread's completion, it is necessary
191     * to wait for the thread to finish before allowing anyone to check the result.
192     */
193    private void waitForThread() {
194        if (this.thread != null && this.thread.isAlive()) {
195            try {
196                this.thread.join();
197            } catch (final InterruptedException e) {
198                this.plugin.getLogger().log(Level.SEVERE, null, e);
199            }
200        }
201    }
202
203    /**
204     * Save an update from dev.bukkit.org into the server's update folder.
205     */
206    private void saveFile() {
207        final File folder = new File(plugin.getDataFolder().getParent(), plugin.getServer().getUpdateFolder());
208        if (!folder.exists() && !folder.mkdir()) {
209            plugin.getLogger().warning("Couldn't create update folder");
210            return;
211        }
212
213        downloadFile();
214    }
215
216    /**
217     * Download a file and save it to the specified folder.
218     */
219    private void downloadFile() {
220        BufferedInputStream inputstream = null;
221        FileOutputStream fout = null;
222        try {
223            final URL fileUrl = new URL(this.versionLink);
224            inputstream = new BufferedInputStream(fileUrl.openStream());
225            fout = new FileOutputStream(new File(this.updateFolder, file.getName()));
226
227            ByteStreams.copy(inputstream, fout);
228        } catch (Exception ex) {
229            this.plugin.getLogger().log(Level.WARNING
230                    , "The auto-updater tried to download a new update, but was unsuccessful.", ex);
231            this.result = Updater.UpdateResult.FAIL_DOWNLOAD;
232        } finally {
233            if (fout != null) {
234                try {
235                    fout.close();
236                } catch (IOException ex) {
237                    this.plugin.getLogger().log(Level.SEVERE, null, ex);
238                }
239            }
240
241            if (inputstream != null) {
242                try {
243                    inputstream.close();
244                } catch (IOException ex) {
245                    this.plugin.getLogger().log(Level.SEVERE, null, ex);
246                }
247            }
248        }
249    }
250
251    /**
252     * Check to see if the program should continue by evaluating whether the plugin is already updated, or shouldn't be
253     * updated.
254     *
255     * @return true if the version was located and is not the same as the remote's newest.
256     */
257    private boolean versionCheck() {
258        final String title = this.versionName;
259        final String localVersion = this.plugin.getDescription().getVersion();
260        final String[] split = title.split(DELIMETER);
261        if (split.length == 2) {
262            // Get the newest file's version number
263            final String remoteVersion = split[1].split(" ")[0];
264
265            if (this.hasTag(localVersion) || !this.shouldUpdate(localVersion, remoteVersion)) {
266                // We already have the latest version, or this build is tagged for no-update
267                this.result = Updater.UpdateResult.NO_UPDATE;
268                return false;
269            }
270        } else {
271            // The file's name did not contain the string 'vVersion'
272            this.plugin.getLogger().warning("File versions should follow the format 'PluginName vVERSION'");
273            this.plugin.getLogger().warning("Please notify the author of this error.");
274            this.result = Updater.UpdateResult.FAIL_NOVERSION;
275            return false;
276        }
277
278        return true;
279    }

281    /**
282     * <b>If you wish to run mathematical versioning checks, edit this method.</b>
283     * <p>
284     * With default behavior, Updater will NOT verify that a remote version available on BukkitDev
285     * which is not this version is indeed an "update".
286     * If a version is present on BukkitDev that is not the version that is currently running,
287     * Updater will assume that it is a newer version.
288     * This is because there is no standard versioning scheme, and creating a calculation that can
289     * determine whether a new update is actually an update is sometimes extremely complicated.
290     * </p>
291     * <p>
292     * Updater will call this method from {@link #versionCheck()} before deciding whether
293     * the remote version is actually an update.
294     * If you have a specific versioning scheme with which a mathematical determination can
295     * be reliably made to decide whether one version is higher than another, you may
296     * revise this method, using the local and remote version parameters, to execute the
297     * appropriate check.
298     * </p>
299     * <p>
300     * Returning a value of <b>false</b> will tell the update process that this is NOT a new version.
301     * Without revision, this method will always consider a remote version at all different from
302     * that of the local version a new update.
303     * </p>
304     * @param localVersion the current version
305     * @param remoteVersion the remote version
306     * @return true if Updater should consider the remote version an update, false if not.
307     */
308    public boolean shouldUpdate(String localVersion, String remoteVersion) {
309        return !localVersion.equalsIgnoreCase(remoteVersion);
310    }
311
312    /**
313     * Evaluate whether the version number is marked showing that it should not be updated by this program.
314     *
315     * @param version a version number to check for tags in.
316     * @return true if updating should be disabled.
317     */
318    private boolean hasTag(String version) {
319        for (String string : Updater.NO_UPDATE_TAG) {
320            if (version.contains(string)) {
321                return true;
322            }
323        }
324
325        return false;
326    }
327
328    /**
329     * Make a connection to the BukkitDev API and request the newest file's details.
330     *
331     * @return true if successful.
332     */
333    private boolean read() {
334        if (this.url == null) {
335            return false;
336        }
337
338        try {
339            final URLConnection conn = this.url.openConnection();
340            conn.setConnectTimeout(5000);
341            conn.setReadTimeout(10000);
342
343            final String userAgent = String.format("%s/v%s (by %s)"
344                    , plugin.getName()
345                    , plugin.getDescription().getVersion()
346                    , plugin.getDescription().getAuthors());
347            conn.addRequestProperty("User-Agent", userAgent);
348
349            final InputStreamReader streamReader = new InputStreamReader(conn.getInputStream(), Charsets.UTF_8);
350
351            final JSONArray array = (JSONArray) JSONValue.parse(streamReader);
352            if (array.isEmpty()) {
353                this.plugin.getLogger().log(Level.WARNING
354                        , "The updater could not find any file for the project id: {0}", this.id);
355                this.result = UpdateResult.FAIL_BADID;
356                return false;
357            }
358
359            //gets the last entry
360            final JSONObject latestUpdate = (JSONObject) array.get(array.size() - 1);
361            this.versionName = (String) latestUpdate.get(Updater.TITLE_VALUE);
362            this.versionLink = (String) latestUpdate.get(Updater.LINK_VALUE);
363
364            return true;
365        } catch (IOException e) {
366            this.plugin.getLogger().severe("The updater could not contact dev.bukkit.org for updating.");
367            this.plugin.getLogger().severe("The site may be experiencing temporary downtime.");
368            this.plugin.getLogger().log(Level.SEVERE, null, e);
369            this.result = UpdateResult.FAIL_DBO;
370            return false;
371        }
372    }

374    /**
375     * Called on main thread when the Updater has finished working, regardless
376     * of result.
377     */
378    public interface UpdateCallback {
379        /**
380         * Called when the updater has finished working.
381         * @param updater The updater instance
382         */
383        void onFinish(Updater updater);
384    }
385
386    private void runUpdater() {
387        if (this.read() && this.versionCheck()) {
388            // Obtain the results of the project's file feed
389            if (this.versionLink != null && this.shouldDownload) {
390                this.saveFile();
391            } else {
392                this.result = UpdateResult.UPDATE_AVAILABLE;
393            }
394        }
395
396        if (this.callback != null) {
397            this.plugin.getServer().getScheduler().runTask(plugin, new Runnable() {
398
399                @Override
400                public void run() {
401                    runCallback();
402                }
403            });
404        }
405    }
406
407    private void runCallback() {
408        this.callback.onFinish(this);
409    }
