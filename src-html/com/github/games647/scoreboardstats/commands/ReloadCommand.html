001package com.github.games647.scoreboardstats.commands;
002
003import com.github.games647.scoreboardstats.Lang;
004import com.github.games647.scoreboardstats.ScoreboardStats;
005
006import org.bukkit.command.CommandSender;
007
008/**
009 * Reload the plugin and the components
010 */
011public class ReloadCommand extends CommandHandler {
012
013    public ReloadCommand(ScoreboardStats plugin) {
014        super("reload", plugin);
015    }
016
017    @Override
018    public void onCommand(CommandSender sender, String subCommand, String... args) {
019        plugin.onReload();
020        sender.sendMessage(Lang.get("onReload"));
021    }
022}
