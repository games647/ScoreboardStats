001package com.github.games647.scoreboardstats.variables.defaults;
002
003import com.github.games647.scoreboardstats.Version;
004import com.github.games647.scoreboardstats.variables.ReplaceEvent;
005import com.github.games647.scoreboardstats.variables.UnsupportedPluginException;
006import com.github.games647.scoreboardstats.variables.VariableReplaceAdapter;
007
008import net.milkbowl.vault.economy.Economy;
009
010import org.bukkit.Bukkit;
011import org.bukkit.entity.Player;
012import org.bukkit.plugin.Plugin;
013import org.bukkit.plugin.RegisteredServiceProvider;
014import org.bukkit.util.NumberConversions;
015
016/**
017 * Replace the economy variable with Vault.
018 *
019 * @see Economy
020 */
021public class VaultVariables extends VariableReplaceAdapter<Plugin> {
022
023    private final Economy economy;
024
025    /**
026     * Creates a new vault replacer
027     */
028    public VaultVariables() {
029        super(Bukkit.getPluginManager().getPlugin("Vault"), "money");
030
031        checkVersion();
032
033        final RegisteredServiceProvider<Economy> economyProvider = Bukkit
034                .getServicesManager().getRegistration(Economy.class);
035        if (economyProvider == null) {
036            //check if an economy plugin is installed otherwise it would throw a exception if the want to replace
037            throw new UnsupportedPluginException("Cannot find an economy plugin");
038        } else {
039            economy = economyProvider.getProvider();
040        }
041    }
042
043    @Override
044    public void onReplace(Player player, String variable, ReplaceEvent replaceEvent) {
045        replaceEvent.setScore(NumberConversions.round(economy.getBalance(player, player.getWorld().getName())));
046    }
047
048    /**
049     * Check if the server has Vault above 1.4.1 installed, because there they
050     * introduced UUID support, but this doesn't make Vault incompatible with
051     * older Minecraft versions
052     *
053     * @see Economy#getBalance(org.bukkit.OfflinePlayer)
054     */
055    private void checkVersion() {
056        final String version = getPlugin().getDescription().getVersion();
057        int end = version.indexOf('-');
058        if (end == -1) {
059            end = version.length();
060        }
061
062        final String cleanVersion = version.substring(0, end);
063        if (Version.compare("1.4.1", cleanVersion) < 0) {
064            throw new UnsupportedPluginException("You have an outdated version of Vault. Please update it");
065        }
066    }
067}
