001package com.github.games647.scoreboardstats.variables.defaults;
002
003import com.github.games647.scoreboardstats.variables.ReplaceEvent;
004import com.github.games647.scoreboardstats.variables.VariableReplaceAdapter;
005
006import org.bukkit.entity.Player;
007import org.bukkit.event.Listener;
008import org.bukkit.inventory.ItemStack;
009import org.bukkit.plugin.Plugin;
010import org.bukkit.util.NumberConversions;
011
012/**
013 * Replace all Bukkit variables
014 */
015public class BukkitVariables extends VariableReplaceAdapter<Plugin> implements Listener {
016
017    private static final int MINUTE_TO_SECOND = 60;
018
019    public BukkitVariables() {
020        super(null, "health", "lifetime", "exp", "no_domage_ticks", "xp_to_Level", "last_damage"
021                , "helmet", "boots", "leggings", "chestplate", "time");
022    }
023
024    @Override
025    public void onReplace(Player player, String variable, ReplaceEvent replaceEvent) {
026        if ("health".equals(variable)) {
027            replaceEvent.setScore(NumberConversions.round(player.getHealth()));
028            return;
029        }
030
031        if ("lifetime".equals(variable)) {
032            // --> Minutes
033            replaceEvent.setScore(player.getTicksLived() / (20 * MINUTE_TO_SECOND));
034            return;
035        }
036
037        if ("exp".equals(variable)) {
038            replaceEvent.setScore(player.getTotalExperience());
039            return;
040        }
041
042        if ("no_damage_ticks".equals(variable)) {
043            // --> Minutes
044            replaceEvent.setScore(player.getNoDamageTicks() / (20 * MINUTE_TO_SECOND));
045            return;
046        }
047
048        if ("xp_to_level".equals(variable)) {
049            replaceEvent.setScore(player.getExpToLevel());
050            return;
051        }
052
053        if ("last_damage".equals(variable)) {
054            //casting should be made after division
055            // --> Minutes
056            replaceEvent.setScore((int) (player.getLastDamage() / (20 * MINUTE_TO_SECOND)));
057            return;
058        }
059
060        if ("helmet".equals(variable)) {
061            replaceEvent.setScore(calculateDurability(player.getInventory().getHelmet()));
062            return;
063        }
064
065        if ("boots".equals(variable)) {
066            replaceEvent.setScore(calculateDurability(player.getInventory().getBoots()));
067            return;
068        }
069
070        if ("leggings".equals(variable)) {
071            replaceEvent.setScore(calculateDurability(player.getInventory().getLeggings()));
072            return;
073        }
074
075        if ("chestplate".equals(variable)) {
076            replaceEvent.setScore(calculateDurability(player.getInventory().getChestplate()));
077            return;
078        }
079
080        if ("time".equals(variable)) {
081            replaceEvent.setScore((int) player.getWorld().getTime());
082        }
083    }
084
085    private int calculateDurability(ItemStack item) {
086        //Check if the user have an item on the slot and if the item isn't a stone block or something
087        if (item == null || item.getType().getMaxDurability() == 0) {
088            return 0;
089        } else {
090            //calculate in percent
091            return item.getDurability() * 100 / item.getType().getMaxDurability();
092        }
093    }
094}
