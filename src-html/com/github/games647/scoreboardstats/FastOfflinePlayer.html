001package com.github.games647.scoreboardstats;
002
003import com.google.common.base.Charsets;
004import com.google.common.base.Objects;
005import com.google.common.collect.Maps;
006
007import java.util.Map;
008import java.util.UUID;
009
010import org.bukkit.Bukkit;
011import org.bukkit.Location;
012import org.bukkit.OfflinePlayer;
013import org.bukkit.entity.Player;
014
015/**
016 * Since 1.7 Bukkit introduced CraftServer.getOfflinePlayer(String name).
017 *
018 * This method is blocking as it looks up the cache. So it could cause lags
019 * on big servers. Because we only need the name this class is made.
020 *
021 * Bukkit already implemented methods to get around this problem, but the
022 * scoreboard team isn't yet modified for methods, which can be called just
023 * with a string.
024 *
025 * Servers <b>below 1.7</b> with use of this class doesn't need to search for the
026 * player.
027 *
028 * @see org.bukkit.Bukkit#getOfflinePlayer(java.lang.String)
029 */
030public class FastOfflinePlayer implements OfflinePlayer {
031
032    private final String playerName;
033
034    /**
035     * Creates a new instance based of this name
036     *
037     * @param playerName the player name
038     */
039    public FastOfflinePlayer(String playerName) {
040        this.playerName = playerName;
041    }
042
043    @Override
044    public boolean isOnline() {
045        //it's a fake player, so it won't be online
046        return false;
047    }
048
049    @Override
050    public String getName() {
051        return playerName;
052    }
053
054    @Override
055    public UUID getUniqueId() {
056        return UUID.nameUUIDFromBytes(playerName.getBytes(Charsets.UTF_8));
057    }
058
059    @Override
060    public boolean isBanned() {
061        return false;
062    }
063
064    @Override
065    public void setBanned(boolean banned) {
066        //ignore
067    }
068
069    @Override
070    public boolean isWhitelisted() {
071        return false;
072    }
073
074    @Override
075    public void setWhitelisted(boolean value) {
076        //ignore
077    }
078
079    @Override
080    public Player getPlayer() {
081        return null;
082    }
083
084    @Override
085    public long getFirstPlayed() {
086        return System.currentTimeMillis();
087    }
088
089    @Override
090    public long getLastPlayed() {
091        return System.currentTimeMillis();
092    }
093
094    @Override
095    public boolean hasPlayedBefore() {
096        return false;
097    }
098
099    @Override
100    public Location getBedSpawnLocation() {
101        return new Location(Bukkit.getWorlds().get(0), 0, 0, 0);
102    }
103
104    @Override
105    public boolean isOp() {
106        return false;
107    }
108
109    @Override
110    public void setOp(boolean value) {
111        //ignore
112    }
113
114    @Override
115    public Map<String, Object> serialize() {
116        final Map<String, Object> result = Maps.newLinkedHashMap();
117
118        result.put("UUID", getUniqueId().toString());
119        result.put("name", playerName);
120
121        return result;
122    }
123
124    @Override
125    public int hashCode() {
126        return Objects.hashCode(playerName);
127    }
128
129    @Override
130    public boolean equals(Object obj) {
131        //check for null too
132        if (!(obj instanceof FastOfflinePlayer)) {
133            return false;
134        }
135
136        final FastOfflinePlayer other = (FastOfflinePlayer) obj;
137        return Objects.equal(playerName, other.playerName);
138    }
139
140    @Override
141    public String toString() {
142        return Objects.toStringHelper(this)
143                .add("playerName", playerName)
144                .toString();
145    }
146}
