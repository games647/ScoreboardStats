001package com.github.games647.scoreboardstats.pvpstats;
002
003import java.lang.ref.WeakReference;
004
005import org.apache.commons.lang.builder.ToStringBuilder;
006import org.bukkit.entity.Player;
007import org.bukkit.metadata.FixedMetadataValue;
008import org.bukkit.plugin.Plugin;
009import org.bukkit.scheduler.BukkitRunnable;
010
011/**
012 * This class is used for loading the player stats.
013 */
014public class StatsLoader implements Runnable {
015
016    protected final Plugin plugin;
017
018    private final boolean uuidSearch;
019    private final WeakReference<Player> weakPlayer;
020    private final WeakReference<Database> weakDatabase;
021
022    /**
023     * Creates a new loader for a specific player
024     *
025     * @param plugin the owning plugin to reschedule
026     * @param uuidSearch should it be searched by uuid
027     * @param player player instance
028     * @param statsDatabase the pvp database
029     */
030    public StatsLoader(Plugin plugin, boolean uuidSearch, Player player, Database statsDatabase) {
031        this.plugin = plugin;
032        this.uuidSearch = uuidSearch;
033
034        //don't prevent the garbage collection of this player if he logs out
035        this.weakPlayer = new WeakReference<Player>(player);
036        this.weakDatabase = new WeakReference<Database>(statsDatabase);
037    }
038
039    @Override
040    public void run() {
041        final Player player = weakPlayer.get();
042        final Database statsDatabase = weakDatabase.get();
043        if (player != null && statsDatabase != null) {
044            final PlayerStats stats = statsDatabase.loadAccount(player);
045            //update player name on every load, because it's changeable
046            stats.setPlayername(player.getName());
047            if (uuidSearch) {
048                //Set the uuid if the server is uuid compatible
049                stats.setUuid(player.getUniqueId());
050            }
051
052            new BukkitRunnable() {
053
054                @Override
055                public void run() {
056                    //possible not thread-safe, so reschedule it while setMetadata is thread-safe
057                    if (player.isOnline()) {
058                        //sets it only if the player is only
059                        player.setMetadata("player_stats", new FixedMetadataValue(plugin, stats));
060                    }
061                }
062            }.runTask(plugin);
063        }
064    }
065
066    @Override
067    public String toString() {
068        return ToStringBuilder.reflectionToString(this);
069    }
070}
