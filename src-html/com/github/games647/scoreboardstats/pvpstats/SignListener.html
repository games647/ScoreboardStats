001package com.github.games647.scoreboardstats.pvpstats;
002
003import de.blablubbabc.insigns.SimpleChanger;
004
005import org.bukkit.Location;
006import org.bukkit.entity.Player;
007import org.bukkit.plugin.Plugin;
008
009/**
010 * Replace some variables on signs with the player individual stats.
011 * The variables will be replaced dynamically
012 *
013 * @see Database
014 */
015public class SignListener extends SimpleChanger {
016
017    private final Database statsDatabase;
018    private final String variable;
019
020    public SignListener(Plugin plugin, String key, Database statsDatabase) {
021        super(plugin, key, plugin.getName().toLowerCase() + ".sign");
022
023        this.variable = key.replace("[", "").replace("]", "");
024        this.statsDatabase = statsDatabase;
025    }
026
027    @Override
028    public String getValue(Player player, Location lctn, String string) {
029        final PlayerStats playerCache = statsDatabase.getCachedStats(player);
030        if (playerCache == null) {
031            //The stats aren't loaded yet
032            return "Not loaded";
033        }
034
035        if ("Kill".equals(variable)) {
036            return Integer.toString(playerCache.getKills());
037        } else if ("Death".equals(variable)) {
038            return Integer.toString(playerCache.getDeaths());
039        } else if ("KDR".equals(variable)) {
040            return Integer.toString(playerCache.getKdr());
041        } else if ("Streak".equals(variable)) {
042            return Integer.toString(playerCache.getKillstreak());
043        }
044
045        return Integer.toString(playerCache.getMobkills());
046    }
047}
